1. Round Robin (RR) Scheduling
#include <stdio.h>
#include <stdlib.h>
#define MAX 10
int queue[MAX], front = -1, rear = -1;
void enqueue(int x) { if(rear == MAX-1) return; if(front==-1) front=0; queue[++rear]=x; }
int dequeue() { if(front>rear) return -1; return queue[front++]; }
void gantt(int n, int bt[], int qt, int at[]) {
    printf("
Gantt Chart:
");
    int time=0, complete=0;
    int rem_bt[MAX]; for(int i=0;i<n;i++) rem_bt[i]=bt[i];
    while(complete<n) {
        int done=1;
        for(int i=0;i<n;i++) {
            if(at[i]<=time && rem_bt[i]>0) {
                done=0;
                int run = (rem_bt[i]>qt)?qt:rem_bt[i];
                for(int j=0;j<run;j++) { printf(" P%d", i); }
                rem_bt[i]-=run;
                time+=run;
                if(rem_bt[i]==0) complete++;
            }
        }
        if(done) { printf(" |"); time++; }
    }
    printf("
");
}

2. Shortest Job First (SJF) Non-Preemptive
#include <stdio.h>
#include <stdlib.h>
typedef struct { int pid, at, bt; } proc;
void swap(proc* a, proc* b) { proc t=*a; *a=*b; *b=t; }
void sort(proc p[], int n) { for(int i=0;i<n-1;i++) for(int j=0;j<n-i-1;j++) if(p[j].at > p[j+1].at) swap(&p[j],&p[j+1]); }
void gantt(proc p[], int n) {
    printf("
Gantt Chart:
");
    int time=0, complete=0;
    int rem[n]; for(int i=0;i<n;i++) rem[i]=p[i].bt;
    while(complete<n) {
        int min_idx=-1, min_bt=9999;
        for(int i=0;i<n;i++) if(p[i].at <= time && rem[i]>0 && rem[i]<min_bt) { min_bt=rem[i]; min_idx=i; }
        if(min_idx!=-1) {
            printf(" P%d", p[min_idx].pid);
            time += rem[min_idx];
            complete++;
            rem[min_idx]=0;
        } else time++;
    }
}
int main() {
    int n; printf("Processes: "); scanf("%d",&n);
    proc p[n]; for(int i=0;i<n;i++) { p[i].pid=i; printf("AT P%d: ",i); scanf("%d",&p[i].at); printf("BT: "); scanf("%d",&p[i].bt); }
    sort(p,n);
    gantt(p,n);
    return 0;
}

3. Dining Philosophers (Semaphores, pthreads)
#include <stdio.h>
#include <pthread.h>
#include <semaphore.h>
#include <unistd.h>
#define N 5
sem_t room, chop[N];
void* philosopher(void* num) {
    int phil = *(int*)num;
    sem_wait(&room);
    printf("
P%d entered room", phil);
    sem_wait(&chop[phil]);
    sem_wait(&chop[(phil+1)%N]);
    printf("
P%d eating", phil);
    sleep(2);
    printf("
P%d finished", phil);
    sem_post(&chop[(phil+1)%N]);
    sem_post(&chop[phil]);
    sem_post(&room);
    return NULL;
}
int main() {
    pthread_t tid[N]; int a[N]; sem_init(&room,0,4);
    for(int i=0;i<N;i++) { sem_init(&chop[i],0,1); a[i]=i; pthread_create(&tid[i],NULL,philosopher,&a[i]); }
    for(int i=0;i<N;i++) pthread_join(tid[i],NULL);
    return 0;
}
Output visual: Sequential eating/thinking avoids deadlock via room limit �.
4. Banker's Algorithm
#include <stdio.h>
int main() {
    int m, n, i, j, k; int need[10][10], alloc[10][10], max[10][10], avail[10];
    printf("Resources m: "); scanf("%d",&m); printf("Processes n: "); scanf("%d",&n);
    printf("Allocation Matrix:
"); for(i=0;i<n;i++) for(j=0;j<m;j++) scanf("%d",&alloc[i][j]);
    printf("Max Matrix:
"); for(i=0;i<n;i++) for(j=0;j<m;j++) scanf("%d",&max[i][j]);
    printf("Available: "); for(i=0;i<m;i++) scanf("%d",&avail[i]);
    for(i=0;i<n;i++) for(j=0;j<m;j++) need[i][j] = max[i][j] - alloc[i][j];
    int finish[10]={0}, safe[10], ind=0;
    int work[10]; for(i=0;i<m;i++) work[i]=avail[i];
    while(ind<n) {
        int f=0; for(i=0;i<n;i++) {
            if(finish[i]==0) {
                f=0; for(j=0;j<m;j++) if(need[i][j]>work[j]) f=1;
                if(!f) { for(j=0;j<m;j++) work[j]+=alloc[i][j]; safe[ind++]=i; finish[i]=1; }
            }
        } if(ind==0) { printf("Unsafe
"); return 0; }
    }
    printf("Safe Sequence: "); for(i=0;i<n;i++) printf("P%d ",safe[i]);
    return 0;
}

6. LRU Page Replacement
#include <stdio.h>
int main() {
    int f,n,i,j,k,pf=0, frame[10], rs[30], count[10]={0};
    printf("Frames: "); scanf("%d",&f); printf("Ref len: "); scanf("%d",&n);
    printf("Ref: "); for(i=0;i<n;i++) scanf("%d",&rs[i]);
    for(i=0;i<f;i++) frame[i]=-1;
    printf("
Frames:
");
    for(i=0;i<n;i++) {
        int found=0; for(j=0;j<f;j++) if(frame[j]==rs[i]) { found=1; count[j]=i; break; }
        if(!found) { int lru=0,minc=count[0]; for(j=1;j<f;j++) if(count[j]<minc) { minc=count[j]; lru=j; }
            frame[lru]=rs[i]; count[lru]=i; pf++;
        }
        for(k=0;k<f;k++) printf("%4d",frame[k]); printf(" %s
",!found?"PF":"Hit");
    }
    printf("PF: %d
",pf);
    return 0;
}
Tracks recent use via counters �.

9. SCAN Disk Scheduling
#include <stdio.h>
int main() {
    int rq[100], head, n, total=0, i,j,max,direction=1;
    printf("Max cyl: "); scanf("%d",&max); printf("Requests: "); scanf("%d",&n); for(i=0;i<n;i++) scanf("%d",&rq[i]);
    printf("Head: "); scanf("%d",&head); printf("Dir (1 right/0 left): "); scanf("%d",&direction);
    printf("Seek: %d",head);
    for(i=0;i<n;i++) {
        for(j=0;j<n-i-1;j++) if(rq[j]>rq[j+1]) { int t=rq[j]; rq[j]=rq[j+1]; rq[j+1]=t; }
    }
    int pos; for(pos=0;rq[pos]<head;pos++);
    if(direction==1) {
        for(i=pos;i<n;i++) { total+=abs(rq[i]-head); printf(" -> %d",rq[i]); head=rq[i]; }
        total+= (max-1 - head); head=max-1;
        for(i=pos-1;i>=0;i--) { total+=abs(rq[i]-head); printf(" -> %d",rq[i]); head=rq[i]; }
    }
    printf("
Total Seek: %d
",total);
    return 0;
}
Elevator to end, then reverse �.
