#include <stdio.h>
#define V 5
#define INF 9999

void runPrim() {
    int g[V][V], key[V], vis[V]={0}, parent[V];

    printf("Enter matrix:\n");
    for(int i=0;i<V;i++)
        for(int j=0;j<V;j++)
            scanf("%d",&g[i][j]);

    for(int i=0;i<V;i++) key[i]=INF;
    key[0]=0; parent[0]=-1;

    for(int c=0;c<V-1;c++) {
        int min=INF,u;

        for(int i=0;i<V;i++)
            if(!vis[i] && key[i]<min)
                min=key[i],u=i;

        vis[u]=1;

        for(int v=0;v<V;v++)
            if(g[u][v] && !vis[v] && g[u][v]<key[v])
                parent[v]=u, key[v]=g[u][v];
    }

    for(int i=1;i<V;i++)
        printf("%d-%d = %d\n",parent[i],i,g[i][parent[i]]);
}

int main() {
    int ch;
    do {
        printf("\n1. Prim\n0. Exit\n");
        scanf("%d",&ch);

        if(ch==1) runPrim();

    } while(ch!=0);
}
